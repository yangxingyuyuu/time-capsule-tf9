  <script>

    const gallery = document.getElementById('gallery');

    const jarBody = document.getElementById('jarBody');


    /* 读取并渲染 */

    function loadImages(filterName = '') {

      const images = JSON.parse(localStorage.getItem('timeCapsuleImages') || '[]');

      gallery.innerHTML = '';

      jarBody.innerHTML = '';


      images

        .filter(item => item.name.includes(filterName))

        .forEach(() => addMiniCapsule());


      images

        .filter(item => item.name.includes(filterName))

        .forEach(item => {

          const card = document.createElement('div');

          card.className = 'photo-card';


          const img = document.createElement('img');

          img.src = item.data;


          const info = document.createElement('div');

          info.className = 'info';

          info.textContent = `${item.name} · ${item.time}`;


          card.appendChild(img);

          card.appendChild(info);

          gallery.appendChild(card);

        });

    }


    /* 保存图片 */

    function saveImages() {

      const name = document.getElementById('studentName').value.trim();

      const input = document.getElementById('imageInput');


      if (!name || !input.files.length) {

        alert('请输入学生姓名并选择图片');

        return;

      }


      const images = JSON.parse(localStorage.getItem('timeCapsuleImages') || '[]');

      let processed = 0;


      Array.from(input.files).forEach(file => {

        const reader = new FileReader();

        reader.onload = e => {

          images.push({

            name,

            data: e.target.result,

            time: new Date().toLocaleString()

          });


          processed++;

          if (processed === input.files.length) {

            localStorage.setItem('timeCapsuleImages', JSON.stringify(images));

            loadImages();

            input.value = '';

            document.getElementById('studentName').value = '';

          }

        };

        reader.readAsDataURL(file);

      });

    }


    /* 添加小胶囊 */

    function addMiniCapsule() {

      const mini = document.createElement('div');

      mini.className = 'mini-capsule';

      mini.style.left = Math.random() * 260 + 'px';

      jarBody.appendChild(mini);

    }


    /* 搜索 */

    document.getElementById('searchInput').addEventListener('input', e => {

      loadImages(e.target.value.trim());

    });


    loadImages();
